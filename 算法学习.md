## 采用欧几里得算法求两个数的公约数
```C
//递归方法
int RGcd(int m, int n)
{
	if (m == 0)
	{
		return n;
	}
	return RGcd(n % m, m);
}

int Gcm(int m, int n)
{
	if (m > n)	//确保m始终小于n
	{
		swap(m, n);
	}
	return RGcd(m, n);
}

//迭代算法
int Gcd(int m, int n)
{
	if (m == 0)
		return n;
	if (n == 0)
		return m;

	if (m > n)
		swap(m, n);

	while (m != 0)
	{
		int c = n % m;
		m = c;
		n = m;
	}
	return n;
}

```
